   260→function AdminVendors() {
   261→  const [statusFilter, setStatusFilter] = useState('all');
   262→  const qc = useQueryClient();
   263→  const [detailVendor, setDetailVendor] = useState<Vendor | null>(null);
   264→  const [rejectDialogOpen, setRejectDialogOpen] = useState(false);
   265→  const [rejectVendorId, setRejectVendorId] = useState<string | null>(null);
   266→  const [rejectionReason, setRejectionReason] = useState('');
   267→  const [showWallet, setShowWallet] = useState(false);
   268→
   269→  const { data: walletData } = useQuery({
   270→    queryKey: ['admin-vendor-wallets'],
   271→    queryFn: () => fetch('/api/admin/vendors/wallets').then(r => r.json()).then((r: any) => r.data || []),
   272→  });
   273→
   274→  const params = new URLSearchParams({ limit: '50' });
   275→  if (statusFilter !== 'all') params.set('status', statusFilter);
   276→
   277→  const { data, isLoading } = useQuery({
   278→    queryKey: ['admin-vendors', statusFilter],
   279→    queryFn: () => fetch(`/api/vendors?${params}`).then(r => r.json()).then((r: any) => r),
   280→  });
   281→
   282→  const updateVendor = useMutation({
   283→    mutationFn: ({ id, status, rejectionReason: reason }: { id: string; status: string; rejectionReason?: string }) =>
   284→      fetch(`/api/vendors/${id}`, {
   285→        method: 'PUT',
   286→        headers: { 'Content-Type': 'application/json' },
   287→        body: JSON.stringify({ status, ...(reason ? { rejectionReason: reason } : {}) }),
   288→      }).then(r => r.json()),
   289→    onSuccess: () => {
   290→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   291→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });
   292→      toast.success('Vendor updated');
   293→    },
   294→    onError: () => toast.error('Failed to update vendor'),
   295→  });
   296→
   297→  const handleReject = () => {
   298→    if (!rejectVendorId) return;
   299→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   300→    setRejectDialogOpen(false);
   301→    setRejectVendorId(null);
   302→    setRejectionReason('');
   303→    setDetailVendor(null);
   304→  };
   305→
   306→  const openRejectDialog = (vendorId: string) => {
   307→    setRejectVendorId(vendorId);
   308→    setRejectionReason('');
   309→    setRejectDialogOpen(true);
   310→  };
   311→
   312→  const statusColor: Record<string, string> = {
   313→    PENDING: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   314→    APPROVED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   315→    SUSPENDED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   316→    REJECTED: 'bg-gray-100 text-gray-700 dark:bg-gray-800 dark:text-gray-300',
   317→  };
   318→
   319→  const pendingCount = data?.data?.filter((v: Vendor) => v.status === 'PENDING').length || 0;
   320→
   321→  return (
   322→    <div className="p-6 space-y-6">
   323→      <div className="flex items-center justify-between flex-wrap gap-3">
   324→        <div>
   325→          <h1 className="text-2xl font-bold">Vendor Management</h1>
   326→          <p className="text-muted-foreground text-sm">Manage vendor accounts and applications</p>
   327→        </div>
   328→        {pendingCount > 0 && (
   329→          <Badge className="bg-amber-500 text-white px-3 py-1 text-sm cursor-pointer" onClick={() => setStatusFilter('PENDING')}>
   330→            <Clock size={14} className="mr-1.5" />{pendingCount} Pending Review
   331→          </Badge>
   332→        )}
   333→      </div>
   334→
   335→      <div className="flex gap-2 flex-wrap">
   336→        {['all', 'PENDING', 'APPROVED', 'SUSPENDED', 'REJECTED'].map(s => (
   337→          <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-amber-600' : ''} onClick={() => setStatusFilter(s)}>
   338→            {s === 'all' ? 'All' : s.charAt(0) + s.slice(1).toLowerCase()}
   339→            {s === 'PENDING' && pendingCount > 0 ? ` (${pendingCount})` : ''}
   340→          </Button>
   341→        ))}
   342→      </div>
   343→
   344→      {isLoading ? <div className="space-y-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-20 rounded-xl" />)}</div> :
   345→        <Card>
   346→          <div className="overflow-x-auto">
   347→            <Table>
   348→              <TableHeader>
   349→                <TableRow>
   350→                  <TableHead>Vendor</TableHead>
   351→                  <TableHead className="hidden md:table-cell">Email</TableHead>
   352→                  <TableHead className="hidden lg:table-cell">GST/PAN</TableHead>
   353→                  <TableHead>Products</TableHead>
   354→                  <TableHead className="hidden sm:table-cell">Revenue</TableHead>
   355→                  <TableHead>Status</TableHead>
   356→                  <TableHead className="text-right">Actions</TableHead>
   357→                </TableRow>
   358→              </TableHeader>

   613→function AdminCategories() {
   614→  const qc = useQueryClient();
   615→  const { data: categories, isLoading } = useQuery({
   616→    queryKey: ['categories'],
   617→    queryFn: () => fetch('/api/categories').then(r => r.json()).then((r: any) => r.data || []),
