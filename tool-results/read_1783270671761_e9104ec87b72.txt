    47→function timeAgo(dateStr: string): string {
    48→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
    49→  if (seconds < 60) return 'Just now';
    50→  const minutes = Math.floor(seconds / 60);
    51→  if (minutes < 60) return `${minutes}m ago`;
    52→  const hours = Math.floor(minutes / 60);
    53→  if (hours < 24) return `${hours}h ago`;
    54→  const days = Math.floor(hours / 24);
    55→  if (days < 30) return `${days}d ago`;
    56→  return new Date(dateStr).toLocaleDateString();
    57→}

    59→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    60→  const stars = [];
    61→  for (let i = 1; i <= 5; i++) {
    62→    if (i <= Math.floor(rating)) stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    63→    else if (i - 0.5 <= rating) stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} style={{ clipPath: 'inset(0 50% 0 0)' }} />);
    64→    else stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    65→  }
    66→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    67→}
