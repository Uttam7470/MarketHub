    43→function timeAgo(dateStr: string): string {
    44→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
    45→  if (seconds < 60) return 'Just now';
    46→  const minutes = Math.floor(seconds / 60);
    47→  if (minutes < 60) return `${minutes}m ago`;
    48→  const hours = Math.floor(minutes / 60);
    49→  if (hours < 24) return `${hours}h ago`;
    50→  const days = Math.floor(hours / 24);
    51→  if (days < 30) return `${days}d ago`;
    52→  return new Date(dateStr).toLocaleDateString();
    53→}

   307→  const params = new URLSearchParams({ limit: '50' });

   311→    queryKey: ['admin-vendors', statusFilter],
   312→    queryFn: () => fetch(`/api/vendors?${params}`).then(r => r.json()).then((r: any) => r),
   313→  });
   314→
   315→  const updateVendor = useMutation({
   316→    mutationFn: ({ id, status, rejectionReason: reason }: { id: string; status: string; rejectionReason?: string }) =>
   317→      fetch(`/api/vendors/${id}`, {
   318→        method: 'PUT',
   319→        headers: { 'Content-Type': 'application/json' },
   320→        body: JSON.stringify({ status, ...(reason ? { rejectionReason: reason } : {}) }),
   321→      }).then(r => r.json()),
   322→    onSuccess: () => {
   323→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   324→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });
   325→      toast.success('Vendor updated');
   326→    },
   327→    onError: () => toast.error('Failed to update vendor'),
   328→  });
   329→
   330→  const handleReject = () => {
   331→    if (!rejectVendorId) return;
   332→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   333→    setRejectDialogOpen(false);
   334→    setRejectVendorId(null);
   335→    setRejectionReason('');
   336→    setDetailVendor(null);
   337→  };
   338→
   339→  const openRejectDialog = (vendorId: string) => {
   340→    setRejectVendorId(vendorId);
   341→    setRejectionReason('');
   342→    setRejectDialogOpen(true);
   343→  };
   344→
