    45→const formatCurrency = (price: number) => '₹' + price.toLocaleString('en-IN', { maximumFractionDigits: 0 });
    46→const discountPercent = (price: number, compare?: number | null) => compare ? Math.round(((compare - price) / compare) * 100) : 0;
    47→const badgeColors: Record<string, string> = { BEST_SELLER: 'bg-emerald-500 text-white', NEW_ARRIVAL: 'bg-blue-500 text-white', LIMITED_TIME: 'bg-red-500 text-white', FESTIVAL_OFFER: 'bg-amber-500 text-white' };
    48→const badgeLabels: Record<string, string> = { BEST_SELLER: 'Best Seller', NEW_ARRIVAL: 'New', LIMITED_TIME: 'Limited', FESTIVAL_OFFER: 'Festival' };

    50→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    51→  const stars = [];
    52→  for (let i = 1; i <= 5; i++) {
    53→    if (i <= Math.floor(rating)) {
    54→      stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    55→    } else if (i - 0.5 <= rating) {
    56→      stars.push(<StarHalf key={i} className="fill-amber-400 text-amber-400" size={size} />);
    57→    } else {
    58→      stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    59→    }
    60→  }
    61→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    62→}

    64→function ProductCard({ product }: { product: Product }) {
    65→  const { navigateTo, setSelectedProductId } = useNavigationStore();
    66→  const { addItem } = useCartStore();
    67→  const { toggleItem, hasItem } = useWishlistStore();
    68→  const { toggleItem: toggleCompare, hasItem: hasCompare } = useCompareStore();
    69→  const isWishlisted = hasItem(product.id);
    70→  const isCompared = hasCompare(product.id);
    71→  const discount = discountPercent(product.price, product.compareAtPrice);
    72→  const mainImage = product.images?.[0]?.url;
    73→
    74→  return (
    75→    <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }} whileHover={{ y: -4 }}>
    76→      <Card className="group overflow-hidden border-border/50 hover:shadow-lg transition-all duration-300 cursor-pointer h-full flex flex-col">
    77→        <div className="relative aspect-square overflow-hidden bg-muted/30">
    78→          {mainImage ? (
    79→            <img src={mainImage} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" loading="lazy" />
    80→          ) : (
    81→            <div className="w-full h-full flex items-center justify-center text-muted-foreground"><Package size={48} /></div>
    82→          )}
    83→          <div className="absolute top-2 left-2 flex flex-col gap-1">
    84→            {discount > 0 && <Badge className="bg-red-500 text-white text-xs">{discount}% OFF</Badge>}
    85→            {product.badge && (
    86→              <Badge className={badgeColors[product.badge] || 'bg-gray-500 text-white'}>{badgeLabels[product.badge]}</Badge>
    87→            )}
    88→          </div>
    89→          {product.stock < 5 && product.stock > 0 && <Badge variant="secondary" className="absolute top-2 right-10 text-xs">Low Stock</Badge>}
    90→          {product.stock === 0 && <div className="absolute inset-0 bg-black/50 flex items-center justify-center"><Badge variant="destructive">Out of Stock</Badge></div>}
    91→          <div className="absolute top-2 right-2 flex flex-col gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
    92→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={(e) => { e.stopPropagation(); toggleItem(product.id); toast.success(isWishlisted ? 'Removed from wishlist' : 'Added to wishlist'); }}>
    93→              <Heart size={14} className={isWishlisted ? 'fill-red-500 text-red-500' : ''} />
    94→            </Button>
    95→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={(e) => { e.stopPropagation(); toggleCompare(product.id); toast.success(isCompared ? 'Removed from compare' : 'Added to compare'); }}>
    96→              <GitCompare size={14} className={isCompared ? 'text-orange-500' : ''} />
    97→            </Button>
    98→          </div>
    99→        </div>
   100→        <CardContent className="p-3 flex-1 flex flex-col gap-1.5" onClick={() => { setSelectedProductId(product.id); navigateTo('product-detail'); }}>
   101→          <p className="text-xs text-muted-foreground truncate">{product.vendor?.businessName}</p>
   102→          <h3 className="text-sm font-medium line-clamp-2 leading-tight min-h-[2.5rem]">{product.name}</h3>
   103→          <div className="flex items-center gap-1">
   104→            <StarRating rating={product.rating} size={12} />
   105→            <span className="text-xs text-muted-foreground">({product.reviewCount})</span>
   106→          </div>
   107→          <div className="flex items-baseline gap-2 mt-auto">
   108→            <span className="text-lg font-bold text-foreground">{formatCurrency(product.price)}</span>
   109→            {product.compareAtPrice && product.compareAtPrice > product.price && (
   110→              <span className="text-sm text-muted-foreground line-through">{formatCurrency(product.compareAtPrice)}</span>
   111→            )}
   112→          </div>
   113→        </CardContent>
   114→        <CardFooter className="p-3 pt-0">
   115→          <Button size="sm" className="w-full bg-orange-500 hover:bg-orange-600 text-white" disabled={product.stock === 0} onClick={(e) => { e.stopPropagation(); addItem({ productId: product.id, name: product.name, price: product.price, image: mainImage || '', vendorName: product.vendor?.businessName || '', vendorId: product.vendorId, stock: product.stock }); toast.success('Added to cart!'); }}>
   116→            <ShoppingCart size={14} className="mr-1.5" /> Add to Cart
   117→          </Button>
   118→        </CardFooter>
   119→      </Card>
   120→    </motion.div>
   121→  );
   122→}

   273→function CategoriesNav({ mobile }: { mobile?: boolean }) {
   274→  const { data: categories } = useQuery({ queryKey: ['categories'], queryFn: () => fetch('/api/categories').then(r => r.json()).then((r: ApiResponse<Category[]>) => r.data || []) });
   275→  const { setSelectedCategory, setCustomerView, setMobileMenuOpen } = useNavigationStore();
   276→
   277→  const handleCategoryClick = (slug: string) => {
   278→    setSelectedCategory(slug);
   279→    setCustomerView('products');
   280→    if (mobile) setMobileMenuOpen(false);
   281→  };
   282→
   283→  if (mobile) {
   284→    return categories?.map(cat => (
   285→      <div key={cat.id} className="space-y-1">
   286→        <Button variant="ghost" className="w-full justify-start font-medium" onClick={() => handleCategoryClick(cat.slug)}>{cat.name}</Button>
   287→        {cat.children?.map(child => (
   288→          <Button key={child.id} variant="ghost" className="w-full justify-start pl-8 text-sm text-muted-foreground" onClick={() => handleCategoryClick(child.slug)}>{child.name}</Button>
   289→        ))}
   290→      </div>
   291→    ));
   292→  }
   293→
   294→  return <>
   295→    <Button variant="ghost" size="sm" className="h-10 text-sm font-medium" onClick={() => { setCustomerView('products'); setSelectedCategory(null); }}>All Products</Button>
   296→    {categories?.map(cat => (
   297→      <DropdownMenu key={cat.id}>
   298→        <DropdownMenuTrigger asChild>
   299→          <Button variant="ghost" size="sm" className="h-10 text-sm font-medium hover:text-orange-500">{cat.name}{cat.children?.length ? <ChevronDown size={14} className="ml-1" /> : null}</Button>
   300→        </DropdownMenuTrigger>
   301→        {cat.children?.length ? (
   302→          <DropdownMenuContent>
   303→            {cat.children.map(child => (
   304→              <DropdownMenuItem key={child.id} onClick={() => handleCategoryClick(child.slug)}>{child.name} <Badge variant="secondary" className="ml-auto text-xs">{child._count?.products || 0}</Badge></DropdownMenuItem>
   305→            ))}
   306→          </DropdownMenuContent>
   307→        ) : null}
   308→      </DropdownMenu>
   309→    ))}
   310→  </>;
   311→}
