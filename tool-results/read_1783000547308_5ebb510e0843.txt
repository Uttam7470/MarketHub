   252→function AdminVendors() {
   253→  const [statusFilter, setStatusFilter] = useState('all');
   254→  const qc = useQueryClient();
   255→  const [detailVendor, setDetailVendor] = useState<Vendor | null>(null);
   256→  const [rejectDialogOpen, setRejectDialogOpen] = useState(false);
   257→  const [rejectVendorId, setRejectVendorId] = useState<string | null>(null);
   258→  const [rejectionReason, setRejectionReason] = useState('');
   259→
   260→  const params = new URLSearchParams({ limit: '50' });
   261→  if (statusFilter !== 'all') params.set('status', statusFilter);
   262→
   263→  const { data, isLoading } = useQuery({
   264→    queryKey: ['admin-vendors', statusFilter],
   265→    queryFn: () => fetch(`/api/vendors?${params}`).then(r => r.json()).then((r: any) => r),
   266→  });
   267→
   268→  const updateVendor = useMutation({
   269→    mutationFn: ({ id, status, rejectionReason: reason }: { id: string; status: string; rejectionReason?: string }) =>
   270→      fetch(`/api/vendors/${id}`, {
   271→        method: 'PUT',
   272→        headers: { 'Content-Type': 'application/json' },
   273→        body: JSON.stringify({ status, ...(reason ? { rejectionReason: reason } : {}) }),
   274→      }).then(r => r.json()),
   275→    onSuccess: () => {
   276→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   277→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });
   278→      toast.success('Vendor updated');
   279→    },
   280→    onError: () => toast.error('Failed to update vendor'),
   281→  });
   282→
   283→  const handleReject = () => {
   284→    if (!rejectVendorId) return;
   285→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   286→    setRejectDialogOpen(false);
   287→    setRejectVendorId(null);
   288→    setRejectionReason('');
   289→    setDetailVendor(null);
   290→  };
   291→
   292→  const openRejectDialog = (vendorId: string) => {
   293→    setRejectVendorId(vendorId);
   294→    setRejectionReason('');
   295→    setRejectDialogOpen(true);
   296→  };
   297→
   298→  const statusColor: Record<string, string> = {
   299→    PENDING: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   300→    APPROVED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   301→    SUSPENDED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   302→    REJECTED: 'bg-gray-100 text-gray-700 dark:bg-gray-800 dark:text-gray-300',
   303→  };
   304→
   305→  const pendingCount = data?.data?.filter((v: Vendor) => v.status === 'PENDING').length || 0;
   306→
   307→  return (
   308→    <div className="p-6 space-y-6">
   309→      <div className="flex items-center justify-between flex-wrap gap-3">
   310→        <div>
   311→          <h1 className="text-2xl font-bold">Vendor Management</h1>
   312→          <p className="text-muted-foreground text-sm">Manage vendor accounts and applications</p>
   313→        </div>
   314→        {pendingCount > 0 && (
   315→          <Badge className="bg-amber-500 text-white px-3 py-1 text-sm cursor-pointer" onClick={() => setStatusFilter('PENDING')}>
   316→            <Clock size={14} className="mr-1.5" />{pendingCount} Pending Review
   317→          </Badge>
   318→        )}
   319→      </div>
   320→
   321→      <div className="flex gap-2 flex-wrap">
   322→        {['all', 'PENDING', 'APPROVED', 'SUSPENDED', 'REJECTED'].map(s => (
   323→          <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-amber-600' : ''} onClick={() => setStatusFilter(s)}>
   324→            {s === 'all' ? 'All' : s.charAt(0) + s.slice(1).toLowerCase()}
   325→            {s === 'PENDING' && pendingCount > 0 ? ` (${pendingCount})` : ''}
   326→          </Button>
   327→        ))}
   328→      </div>
   329→
   330→      {isLoading ? <div className="space-y-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-20 rounded-xl" />)}</div> :
   331→        <Card>
   332→          <div className="overflow-x-auto">
   333→            <Table>
   334→              <TableHeader>
   335→                <TableRow>
   336→                  <TableHead>Vendor</TableHead>
   337→                  <TableHead className="hidden md:table-cell">Email</TableHead>
   338→                  <TableHead className="hidden lg:table-cell">GST/PAN</TableHead>
   339→                  <TableHead>Products</TableHead>
   340→                  <TableHead className="hidden sm:table-cell">Revenue</TableHead>
   341→                  <TableHead>Status</TableHead>
   342→                  <TableHead className="text-right">Actions</TableHead>
   343→                </TableRow>
   344→              </TableHeader>
   345→              <TableBody>
   346→                {!data?.data?.length && <TableRow><TableCell colSpan={8} className="text-center py-8 text-muted-foreground">No vendors found</TableCell></TableRow>}
   347→                {data?.data?.map((vendor: Vendor) => (
   348→                  <TableRow key={vendor.id} className={vendor.status === 'PENDING' ? 'bg-amber-50/50 dark:bg-amber-900/5' : ''}>
   349→                    <TableCell>
   350→                      <div className="flex items-center gap-3">

  1381→function AdminFAQ() {
  1382→  const qc = useQueryClient();
  1383→  const [showAdd, setShowAdd] = useState(false);
  1384→  const [editId, setEditId] = useState<string | null>(null);
  1385→  const [form, setForm] = useState({ question: '', answer: '', category: 'General', sortOrder: '0' });
  1386→
  1387→  const { data: faqs, isLoading } = useQuery({
  1388→    queryKey: ['admin-faq'],
  1389→    queryFn: () => fetch('/api/faq').then(r => r.json()).then((r: any) => r.data || []),
  1390→  });
  1391→
  1392→  const saveMutation = useMutation({
  1393→    mutationFn: () => fetch('/api/faq', {
  1394→      method: editId ? 'PUT' : 'POST',
  1395→      headers: { 'Content-Type': 'application/json' },
  1396→      body: JSON.stringify({ ...(editId ? { id: editId } : {}), ...form, sortOrder: parseInt(form.sortOrder), isActive: true }),
  1397→    }).then(r => r.json()),
  1398→    onSuccess: () => { qc.invalidateQueries({ queryKey: ['admin-faq'] }); toast.success(editId ? 'FAQ updated' : 'FAQ added'); setShowAdd(false); setEditId(null); setForm({ question: '', answer: '', category: 'General', sortOrder: '0' }); },
  1399→    onError: () => toast.error('Failed to save FAQ'),
  1400→  });
  1401→
  1402→  const deleteMutation = useMutation({
  1403→    mutationFn: (id: string) => fetch(`/api/faq?id=${id}`, { method: 'DELETE' }).then(r => r.json()),
  1404→    onSuccess: () => { qc.invalidateQueries({ queryKey: ['admin-faq'] }); toast.success('FAQ deleted'); },
  1405→  });
  1406→
  1407→  const handleEdit = (faq: any) => {
  1408→    setForm({ question: faq.question, answer: faq.answer, category: faq.category, sortOrder: String(faq.sortOrder) });
  1409→    setEditId(faq.id);
  1410→    setShowAdd(true);
  1411→  };
  1412→
  1413→  return (
  1414→    <div className="p-6 space-y-6">
  1415→      <div className="flex items-center justify-between">
  1416→        <div><h1 className="text-2xl font-bold">FAQ Management</h1><p className="text-muted-foreground text-sm">Manage frequently asked questions</p></div>
  1417→        <Button className="bg-amber-600 hover:bg-amber-700" onClick={() => { setForm({ question: '', answer: '', category: 'General', sortOrder: '0' }); setEditId(null); setShowAdd(true); }}><Plus size={16} className="mr-1.5" />Add FAQ</Button>
  1418→      </div>
  1419→      {isLoading ? <div className="space-y-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-16 rounded-xl" />)}</div> :
  1420→        <Card className="p-4">
  1421→          <Accordion type="multiple">
  1422→            {(faqs || []).map((faq: any, i: number) => (
  1423→              <AccordionItem key={faq.id} value={faq.id}>
  1424→                <AccordionTrigger className="text-left hover:no-underline">
  1425→                  <div className="flex items-center gap-3 flex-1 pr-4">
  1426→                    <span className="text-sm font-medium flex-1 truncate">{faq.question}</span>
  1427→                    <Badge variant="outline" className="text-xs shrink-0">{faq.category}</Badge>
  1428→                  </div>
  1429→                </AccordionTrigger>
  1430→                <AccordionContent>
  1431→                  <p className="text-sm text-muted-foreground mb-3">{faq.answer}</p>
  1432→                  <div className="flex gap-2">
  1433→                    <Button variant="outline" size="sm" onClick={() => handleEdit(faq)}><Pencil size={12} className="mr-1" />Edit</Button>
  1434→                    <Button variant="outline" size="sm" className="text-destructive" onClick={() => deleteMutation.mutate(faq.id)}><Trash2 size={12} className="mr-1" />Delete</Button>
  1435→                  </div>
  1436→                </AccordionContent>
  1437→              </AccordionItem>
  1438→            ))}
  1439→          </Accordion>
  1440→          {(!faqs || faqs.length === 0) && <p className="text-muted-foreground text-sm text-center py-8">No FAQs yet</p>}
  1441→        </Card>}
  1442→      <Dialog open={showAdd} onOpenChange={setShowAdd}>
  1443→        <DialogContent><DialogHeader><DialogTitle>{editId ? 'Edit FAQ' : 'Add FAQ'}</DialogTitle></DialogHeader>
  1444→          <div className="space-y-4">
  1445→            <div><Label>Question *</Label><Input value={form.question} onChange={e => setForm(f => ({ ...f, question: e.target.value }))} className="mt-1" /></div>

  1448→              <div><Label>Category</Label><Input value={form.category} onChange={e => setForm(f => ({ ...f, category: e.target.value }))} className="mt-1" placeholder="e.g. General, Shipping, Returns" /></div>
