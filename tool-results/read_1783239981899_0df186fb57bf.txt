    52→function useRequireAuth() {
    53→  const { isAuthenticated } = useAuthStore();
    54→  const { navigateTo } = useNavigationStore();
    55→  return (action: string) => {
    56→    if (!isAuthenticated) {
    57→      toast.warning('Login required', { description: `Please log in to ${action}.` });
    58→      navigateTo('login');
    59→      return false;
    60→    }
    61→    return true;
    62→  };
    63→}
    64→const discountPercent = (price: number, compare?: number | null) => compare ? Math.round(((compare - price) / compare) * 100) : 0;
    65→const badgeColors: Record<string, string> = { BEST_SELLER: 'bg-emerald-500 text-white', NEW_ARRIVAL: 'bg-blue-500 text-white', LIMITED_TIME: 'bg-red-500 text-white', FESTIVAL_OFFER: 'bg-amber-500 text-white' };
    66→const badgeLabels: Record<string, string> = { BEST_SELLER: 'Best Seller', NEW_ARRIVAL: 'New', LIMITED_TIME: 'Limited', FESTIVAL_OFFER: 'Festival' };

    68→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    69→  const stars = [];
    70→  for (let i = 1; i <= 5; i++) {
    71→    if (i <= Math.floor(rating)) {
    72→      stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    73→    } else if (i - 0.5 <= rating) {
    74→      stars.push(<StarHalf key={i} className="fill-amber-400 text-amber-400" size={size} />);
    75→    } else {
    76→      stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    77→    }
    78→  }
    79→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    80→}

    82→function ProductCard({ product }: { product: Product }) {
    83→  const { navigateTo, setSelectedProductId } = useNavigationStore();
    84→  const { addItem } = useCartStore();
    85→  const { toggleItem, hasItem } = useWishlistStore();
    86→  const { toggleItem: toggleCompare, hasItem: hasCompare } = useCompareStore();
    87→  const requireAuth = useRequireAuth();
    88→  const isWishlisted = hasItem(product.id);
    89→  const isCompared = hasCompare(product.id);
    90→  const discount = discountPercent(product.price, product.compareAtPrice);
    91→  const mainImage = product.images?.[0]?.url;
    92→
    93→  const handleAddToCart = (e: React.MouseEvent) => {
    94→    e.stopPropagation();
    95→    if (!requireAuth('add items to cart')) return;
    96→    addItem({ productId: product.id, name: product.name, price: product.price, image: mainImage || '', vendorName: product.vendor?.businessName || '', vendorId: product.vendorId, stock: product.stock });
    97→    toast.success('Added to cart', { description: `${product.name} has been added to your cart.` });
    98→  };
    99→
   100→  const handleWishlistToggle = (e: React.MouseEvent) => {
   101→    e.stopPropagation();
   102→    if (!requireAuth('add items to wishlist')) return;
   103→    toggleItem(product.id);
   104→    if (isWishlisted) {
   105→      toast.info('Removed from wishlist');
   106→    } else {
   107→      toast.success('Added to wishlist', { description: product.name });
   108→    }
   109→  };
   110→
   111→  const handleCompareToggle = (e: React.MouseEvent) => {
   112→    e.stopPropagation();
   113→    if (!requireAuth('compare products')) return;
   114→    toggleCompare(product.id);
   115→    if (isCompared) {
   116→      toast.info('Removed from comparison');
   117→    } else {
   118→      toast.success('Added to comparison');
   119→    }
   120→  };
   121→
   122→  return (
   123→    <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }} whileHover={{ y: -4 }}>
   124→      <Card role="article" aria-label={`${product.name}, ${formatCurrency(product.price)}`} className="group overflow-hidden border-border/50 hover:shadow-lg hover:-translate-y-1 transition-all duration-200 cursor-pointer h-full flex flex-col">
   125→        <div className="relative aspect-square overflow-hidden bg-muted/30">
   126→          {mainImage ? (
   127→            <img src={mainImage} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" loading="lazy" />
   128→          ) : (
   129→            <div className="w-full h-full flex items-center justify-center text-muted-foreground"><Package size={48} /></div>
   130→          )}
   131→          <div className="absolute top-2 left-2 flex flex-col gap-1">
   132→            {discount > 0 && <Badge className="bg-red-500 text-white text-xs">{discount}% OFF</Badge>}
   133→            {product.badge && (
   134→              <Badge className={badgeColors[product.badge] || 'bg-gray-500 text-white'}>{badgeLabels[product.badge]}</Badge>
   135→            )}
   136→          </div>
   137→          {product.stock < 5 && product.stock > 0 && <Badge variant="secondary" className="absolute top-2 right-10 text-xs">Low Stock</Badge>}
   138→          {product.stock === 0 && <div className="absolute inset-0 bg-black/50 flex items-center justify-center"><Badge variant="destructive">Out of Stock</Badge></div>}
   139→          <div className="absolute top-2 right-2 flex flex-col gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   140→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" aria-label={isWishlisted ? `Remove ${product.name} from wishlist` : `Add ${product.name} to wishlist`} onClick={handleWishlistToggle}>
   141→              <Heart size={14} className={isWishlisted ? 'fill-red-500 text-red-500' : ''} />
   142→            </Button>
   143→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={handleCompareToggle}>
   144→              <GitCompare size={14} className={isCompared ? 'text-orange-500' : ''} />
   145→            </Button>
   146→          </div>
   147→        </div>
   148→        <CardContent className="p-3 flex-1 flex flex-col gap-1.5" onClick={() => { setSelectedProductId(product.id); navigateTo('product-detail'); }}>
   149→          <p className="text-xs text-muted-foreground truncate">{product.vendor?.businessName}</p>
   150→          <h3 className="text-sm font-medium line-clamp-2 leading-tight min-h-[2.5rem]">{product.name}</h3>
   151→          <div className="flex items-center gap-1">
   152→            <StarRating rating={product.rating} size={12} />
   153→            <span className="text-xs text-muted-foreground">({product.reviewCount})</span>
   154→          </div>
   155→          <div className="flex items-baseline gap-2 mt-auto">
   156→            <span className="text-lg font-bold text-foreground">{formatCurrency(product.price)}</span>
   157→            {product.compareAtPrice && product.compareAtPrice > product.price && (
   158→              <span className="text-sm text-muted-foreground line-through">{formatCurrency(product.compareAtPrice)}</span>
   159→            )}
   160→          </div>
   161→        </CardContent>
   162→        <CardFooter className="p-3 pt-0">
   163→          <Button size="sm" className="w-full bg-orange-500 hover:bg-orange-600 text-white" aria-label={`Add ${product.name} to cart`} disabled={product.stock === 0} onClick={handleAddToCart}>
   164→            <ShoppingCart size={14} className="mr-1.5" /> Add to Cart
   165→          </Button>
   166→        </CardFooter>
   167→      </Card>
   168→    </motion.div>
   169→  );
   170→}

  1109→function ReviewsSection({ productId, reviews, reviewCount, rating }: { productId: string; reviews: any[]; reviewCount: number; rating: number }) {
  1110→  const { isAuthenticated, user } = useAuthStore();

  1113→  const [showForm, setShowForm] = useState(false);
  1114→  const [form, setForm] = useState({ rating: 5, title: '', comment: '', images: '' });
  1115→  const [sortBy, setSortBy] = useState('latest');
  1116→  const sortedReviews = useMemo(() => {
  1117→    const r = [...reviews];
  1118→    switch(sortBy) {
  1119→      case 'highest': return r.sort((a,b) => b.rating - a.rating);
  1120→      case 'lowest': return r.sort((a,b) => a.rating - b.rating);
  1121→      case 'helpful': return r.sort((a,b) => (b.helpfulCount||0) - (a.helpfulCount||0));
  1122→      default: return r.sort((a,b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
  1123→    }
  1124→  }, [reviews, sortBy]);
