    35→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    36→  const stars = [];
    37→  for (let i = 1; i <= 5; i++) {
    38→    if (i <= Math.floor(rating)) stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    39→    else if (i - 0.5 <= rating) stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} style={{ clipPath: 'inset(0 50% 0 0)' }} />);
    40→    else stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    41→  }
    42→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    43→}

   240→function VendorProducts() {
   241→  const { vendorId } = useAuthStore();
   242→  const { setSelectedProductId, setVendorView } = useNavigationStore();
   243→  const qc = useQueryClient();
   244→  const [page, setPage] = useState(1);
   245→  const [search, setSearch] = useState('');
   246→  const [statusFilter, setStatusFilter] = useState<string>('all');
   247→  const [deleteId, setDeleteId] = useState<string | null>(null);
   248→  const [importOpen, setImportOpen] = useState(false);
   249→  const [importFile, setImportFile] = useState<File | null>(null);
   250→  const [importing, setImporting] = useState(false);
   251→
   252→  const { data, isLoading } = useQuery({
   253→    queryKey: ['vendor-products', vendorId, page, search, statusFilter],
   254→    queryFn: () => {
   255→      const params = new URLSearchParams({ vendorId: vendorId || '', page: String(page), limit: '12', search });
   256→      if (statusFilter !== 'all') params.set('status', statusFilter);
   257→      return fetch(`/api/products?${params}`).then(r => r.json()).then((r: ApiResponse<Product[]>) => r);
   258→    },
   259→    enabled: !!vendorId,
   260→  });
   261→
   262→  const deleteMutation = useMutation({
   263→    mutationFn: (id: string) => fetch(`/api/products?id=${id}`, { method: 'DELETE' }).then(r => r.json()),
   264→    onSuccess: () => { qc.invalidateQueries({ queryKey: ['vendor-products'] }); toast.success('Product deleted'); setDeleteId(null); },
   265→    onError: () => toast.error('Failed to delete product'),
   266→  });
   267→
   268→  const duplicateMutation = useMutation({
   269→    mutationFn: (id: string) => fetch(`/api/vendor/products/${id}/duplicate`, { method: 'POST' }).then(r => r.json()),
   270→    onSuccess: (data) => {
   271→      if (data.success) { toast.success('Product duplicated successfully'); qc.invalidateQueries({ queryKey: ['vendor-products'] }); }
   272→      else toast.error(data.error || 'Failed to duplicate');
   273→    },
   274→    onError: () => toast.error('Failed to duplicate product'),
   275→  });
   276→
   277→  const handleBulkImport = async () => {
   278→    if (!importFile || !vendorId) return;
   279→    setImporting(true);
   280→    try {
   281→      const text = await importFile.text();
   282→      const lines = text.split('\n').filter(l => l.trim());
   283→      if (lines.length < 2) { toast.error('CSV file is empty or has no data rows'); setImporting(false); return; }
   284→      const headers = lines[0].split(',').map(h => h.trim().toLowerCase());
   285→      const products = [];
   286→      for (let i = 1; i < lines.length; i++) {
   287→        const values = lines[i].split(',').map(v => v.trim());
   288→        const row: Record<string, string> = {};
   289→        headers.forEach((h, idx) => { row[h] = values[idx] || ''; });
   290→        products.push(row);
   291→      }
   292→      const res = await fetch('/api/vendor/products/bulk-import', {
   293→        method: 'POST',
   294→        headers: { 'Content-Type': 'application/json' },
   295→        body: JSON.stringify({ vendorId, products }),
   296→      });
   297→      const result = await res.json();
   298→      if (result.success) {
   299→        const d = result.data;
   300→        toast.success(`Imported ${d.successCount}/${d.totalProducts} products successfully`);
   301→        if (d.errors.length > 0) toast.error(`${d.errors.length} errors occurred`);
   302→        qc.invalidateQueries({ queryKey: ['vendor-products'] });
   303→        setImportOpen(false);
   304→        setImportFile(null);
   305→      } else {
   306→        toast.error(result.error || 'Import failed');
   307→      }
   308→    } catch {
   309→      toast.error('Failed to parse CSV file');
   310→    }
   311→    setImporting(false);
   312→  };
   313→
   314→  const downloadTemplate = () => {
   315→    const csv = 'name,description,price,compareAtPrice,stock,sku,categoryId,brandId,weight\nSample Product,Sample description,999,1299,50,SKU001,cat_id_here,brand_id_here,0.5\n';
   316→    const blob = new Blob([csv], { type: 'text/csv' });
   317→    const url = URL.createObjectURL(blob);
   318→    const a = document.createElement('a');
   319→    a.href = url; a.download = 'product_import_template.csv'; a.click();
   320→    URL.revokeObjectURL(url);
   321→  };
   322→
   323→  const handleBulkExport = async () => {
   324→    try {
   325→      const params = new URLSearchParams({ vendorId: vendorId || '', page: '1', limit: '9999', search: '' });
   326→      const res = await fetch(`/api/products?${params}`);
   327→      const json = await res.json();
   328→      const products: Product[] = json.data || [];
   329→      if (products.length === 0) { toast.error('No products to export'); return; }
   330→      const headers = ['name', 'sku', 'price', 'stock', 'status', 'category'];
   331→      const rows = products.map(p => [
   332→        `"${p.name}"`, `"${p.sku || ''}"`, String(p.price), String(p.stock),
   333→        p.productStatus, `"${p.category?.name || ''}"`
   334→      ].join(','));
   335→      const csv = [headers.join(','), ...rows].join('\n');
   336→      const blob = new Blob([csv], { type: 'text/csv' });
   337→      const url = URL.createObjectURL(blob);
   338→      const a = document.createElement('a');
   339→      a.href = url; a.download = `products_export_${new Date().toISOString().slice(0, 10)}.csv`; a.click();
   340→      URL.revokeObjectURL(url);
   341→      toast.success(`Exported ${products.length} products`);
   342→    } catch {
   343→      toast.error('Failed to export products');
   344→    }
   345→  };
   346→
   347→  const statusColor: Record<string, string> = {
   348→    PUBLISHED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   349→    DRAFT: 'bg-gray-100 text-gray-700 dark:bg-gray-900/30 dark:text-gray-400',
   350→    ARCHIVED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   351→  };
   352→
   353→  const statusFilters = ['all', 'PUBLISHED', 'DRAFT', 'ARCHIVED'];
   354→
   355→  return (
   356→    <div className="p-6 space-y-6">
   357→      <div className="flex flex-wrap items-center justify-between gap-4">
   358→        <div><h1 className="text-2xl font-bold">Products</h1><p className="text-muted-foreground text-sm">Manage your product catalog</p></div>
   359→        <div className="flex items-center gap-2 flex-wrap">
   360→          <Button className="bg-orange-500 hover:bg-orange-600" onClick={() => setVendorView('vendor-add-product')}><Plus size={16} className="mr-1.5" />Add Product</Button>
   361→          <Button variant="outline" onClick={() => setImportOpen(true)}><Upload size={16} className="mr-1.5" />Bulk Import</Button>
   362→          <Button variant="outline" onClick={handleBulkExport}><Download size={16} className="mr-1.5" />Bulk Export</Button>
   363→        </div>
   364→      </div>
   365→
   366→      <div className="flex flex-col sm:flex-row gap-3">
   367→        <div className="relative flex-1 max-w-md"><Search size={16} className="absolute left-3 top-1/2 -translate-y-1/2 text-muted-foreground" /><Input placeholder="Search products..." value={search} onChange={e => { setSearch(e.target.value); setPage(1); }} className="pl-9" /></div>
   368→        <div className="flex gap-2 flex-wrap">
   369→          {statusFilters.map(s => (
   370→            <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-orange-500 hover:bg-orange-600' : ''} onClick={() => { setStatusFilter(s); setPage(1); }}>
   371→              {s === 'all' ? 'All' : s}
   372→            </Button>
   373→          ))}
   374→        </div>
   375→      </div>
   376→
   377→      {isLoading ? <div className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 gap-4">{Array.from({length: 6}).map((_, i) => <Skeleton key={i} className="h-40 rounded-xl" />)}</div> :
   378→      <Card>
   379→        <Table>
   380→          <TableHeader><TableRow><TableHead>Product</TableHead><TableHead className="hidden md:table-cell">Category</TableHead><TableHead>Price</TableHead><TableHead>Stock</TableHead><TableHead className="hidden sm:table-cell">Status</TableHead><TableHead className="hidden sm:table-cell">Rating</TableHead><TableHead className="text-right">Actions</TableHead></TableRow></TableHeader>

   382→            {!data?.data?.length && <TableRow><TableCell colSpan={7} className="text-center py-8 text-muted-foreground">No products found</TableCell></TableRow>}
   383→            {data?.data?.map(product => (
