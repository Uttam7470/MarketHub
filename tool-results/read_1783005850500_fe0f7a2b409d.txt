   261→function AdminVendors() {
   262→  const [statusFilter, setStatusFilter] = useState('all');
   263→  const qc = useQueryClient();
   264→  const [detailVendor, setDetailVendor] = useState<Vendor | null>(null);
   265→  const [rejectDialogOpen, setRejectDialogOpen] = useState(false);
   266→  const [rejectVendorId, setRejectVendorId] = useState<string | null>(null);
   267→  const [rejectionReason, setRejectionReason] = useState('');
   268→  const [showWallet, setShowWallet] = useState(false);
   269→
   270→  const { data: walletData } = useQuery({
   271→    queryKey: ['admin-vendor-wallets'],
   272→    queryFn: () => fetch('/api/admin/vendors/wallets').then(r => r.json()).then((r: any) => r.data || []),
   273→  });
   274→
   275→  const params = new URLSearchParams({ limit: '50' });
   276→  if (statusFilter !== 'all') params.set('status', statusFilter);
   277→
   278→  const { data, isLoading } = useQuery({
   279→    queryKey: ['admin-vendors', statusFilter],
   280→    queryFn: () => fetch(`/api/vendors?${params}`).then(r => r.json()).then((r: any) => r),
   281→  });
   282→
   283→  const updateVendor = useMutation({
   284→    mutationFn: ({ id, status, rejectionReason: reason }: { id: string; status: string; rejectionReason?: string }) =>
   285→      fetch(`/api/vendors/${id}`, {
   286→        method: 'PUT',
   287→        headers: { 'Content-Type': 'application/json' },
   288→        body: JSON.stringify({ status, ...(reason ? { rejectionReason: reason } : {}) }),
   289→      }).then(r => r.json()),
   290→    onSuccess: () => {
   291→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   292→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });
   293→      toast.success('Vendor updated');
   294→    },
   295→    onError: () => toast.error('Failed to update vendor'),
   296→  });
   297→
   298→  const handleReject = () => {
   299→    if (!rejectVendorId) return;
   300→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   301→    setRejectDialogOpen(false);
   302→    setRejectVendorId(null);
   303→    setRejectionReason('');
   304→    setDetailVendor(null);
   305→  };
   306→
   307→  const openRejectDialog = (vendorId: string) => {
   308→    setRejectVendorId(vendorId);
   309→    setRejectionReason('');
   310→    setRejectDialogOpen(true);
   311→  };
   312→
   313→  const statusColor: Record<string, string> = {
   314→    PENDING: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   315→    APPROVED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   316→    SUSPENDED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   317→    REJECTED: 'bg-gray-100 text-gray-700 dark:bg-gray-800 dark:text-gray-300',
   318→  };
   319→
   320→  const pendingCount = data?.data?.filter((v: Vendor) => v.status === 'PENDING').length || 0;
   321→
   322→  return (
   323→    <div className="p-6 space-y-6">
   324→      <div className="flex items-center justify-between flex-wrap gap-3">
   325→        <div>
   326→          <h1 className="text-2xl font-bold">Vendor Management</h1>
   327→          <p className="text-muted-foreground text-sm">Manage vendor accounts and applications</p>
   328→        </div>
   329→        {pendingCount > 0 && (
   330→          <Badge className="bg-amber-500 text-white px-3 py-1 text-sm cursor-pointer" onClick={() => setStatusFilter('PENDING')}>
   331→            <Clock size={14} className="mr-1.5" />{pendingCount} Pending Review
   332→          </Badge>
   333→        )}
   334→      </div>
   335→
   336→      <div className="flex gap-2 flex-wrap">
   337→        {['all', 'PENDING', 'APPROVED', 'SUSPENDED', 'REJECTED'].map(s => (
   338→          <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-amber-600' : ''} onClick={() => setStatusFilter(s)}>
   339→            {s === 'all' ? 'All' : s.charAt(0) + s.slice(1).toLowerCase()}
   340→            {s === 'PENDING' && pendingCount > 0 ? ` (${pendingCount})` : ''}
   341→          </Button>
   342→        ))}
   343→      </div>
   344→
   345→      {isLoading ? <div className="space-y-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-20 rounded-xl" />)}</div> :
   346→        <Card>
   347→          <div className="overflow-x-auto">
   348→            <Table>
   349→              <TableHeader>
   350→                <TableRow>
   351→                  <TableHead>Vendor</TableHead>
   352→                  <TableHead className="hidden md:table-cell">Email</TableHead>
   353→                  <TableHead className="hidden lg:table-cell">GST/PAN</TableHead>
   354→                  <TableHead>Products</TableHead>
   355→                  <TableHead className="hidden sm:table-cell">Revenue</TableHead>
   356→                  <TableHead>Status</TableHead>
   357→                  <TableHead className="text-right">Actions</TableHead>
   358→                </TableRow>

   614→function AdminCategories() {
   615→  const qc = useQueryClient();
   616→  const { data: categories, isLoading } = useQuery({
   617→    queryKey: ['categories'],
   618→    queryFn: () => fetch('/api/categories').then(r => r.json()).then((r: any) => r.data || []),
   619→  });
   620→
   621→  const [showAdd, setShowAdd] = useState(false);

   623→  const [form, setForm] = useState({ name: '', description: '', parentId: '', sortOrder: '0' });
   624→
   625→  const saveMutation = useMutation({
   626→    mutationFn: async () => {
   627→      if (editId) {
   628→        return fetch('/api/categories', { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id: editId, name: form.name, description: form.description, parentId: form.parentId || null, sortOrder: parseInt(form.sortOrder) }) }).then(r => r.json());

   637→    mutationFn: (id: string) => fetch(`/api/categories?id=${id}`, { method: 'DELETE' }).then(r => r.json()),
   638→    onSuccess: () => { qc.invalidateQueries({ queryKey: ['categories'] }); toast.success('Category deleted'); },
   639→    onError: () => toast.error('Failed to delete'),

   641→
   642→  const handleEdit = (cat: any) => { setForm({ name: cat.name, description: cat.description || '', parentId: cat.parentId || '', sortOrder: String(cat.sortOrder) }); setEditId(cat.id); setShowAdd(true); };

   649→      </div>
