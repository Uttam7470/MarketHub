    46→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    47→  const stars = [];
    48→  for (let i = 1; i <= 5; i++) {
    49→    if (i <= Math.floor(rating)) {
    50→      stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    51→    } else if (i - 0.5 <= rating) {
    52→      stars.push(<StarHalf key={i} className="fill-amber-400 text-amber-400" size={size} />);
    53→    } else {
    54→      stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    55→    }
    56→  }
    57→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    58→}

    60→function ProductCard({ product }: { product: Product }) {
    61→  const { navigateTo, setSelectedProductId } = useNavigationStore();
    62→  const { addItem } = useCartStore();
    63→  const { toggleItem, hasItem } = useWishlistStore();
    64→  const { toggleItem: toggleCompare, hasItem: hasCompare } = useCompareStore();
    65→  const isWishlisted = hasItem(product.id);
    66→  const isCompared = hasCompare(product.id);
    67→  const discount = discountPercent(product.price, product.compareAtPrice);
    68→  const mainImage = product.images?.[0]?.url;
    69→
    70→  return (
    71→    <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }} whileHover={{ y: -4 }}>
    72→      <Card className="group overflow-hidden border-border/50 hover:shadow-lg transition-all duration-300 cursor-pointer h-full flex flex-col">
    73→        <div className="relative aspect-square overflow-hidden bg-muted/30">
    74→          {mainImage ? (
    75→            <img src={mainImage} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" loading="lazy" />
    76→          ) : (
    77→            <div className="w-full h-full flex items-center justify-center text-muted-foreground"><Package size={48} /></div>
    78→          )}
    79→          {discount > 0 && <Badge className="absolute top-2 left-2 bg-red-500 text-white text-xs">{discount}% OFF</Badge>}
    80→          {product.stock < 5 && product.stock > 0 && <Badge variant="secondary" className="absolute top-2 right-10 text-xs">Low Stock</Badge>}
    81→          {product.stock === 0 && <div className="absolute inset-0 bg-black/50 flex items-center justify-center"><Badge variant="destructive">Out of Stock</Badge></div>}
    82→          <div className="absolute top-2 right-2 flex flex-col gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
    83→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={(e) => { e.stopPropagation(); toggleItem(product.id); toast.success(isWishlisted ? 'Removed from wishlist' : 'Added to wishlist'); }}>
    84→              <Heart size={14} className={isWishlisted ? 'fill-red-500 text-red-500' : ''} />
    85→            </Button>
    86→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={(e) => { e.stopPropagation(); toggleCompare(product.id); toast.success(isCompared ? 'Removed from compare' : 'Added to compare'); }}>
    87→              <GitCompare size={14} className={isCompared ? 'text-orange-500' : ''} />
    88→            </Button>
    89→          </div>
    90→        </div>
    91→        <CardContent className="p-3 flex-1 flex flex-col gap-1.5" onClick={() => { setSelectedProductId(product.id); navigateTo('product-detail'); }}>
    92→          <p className="text-xs text-muted-foreground truncate">{product.vendor?.businessName}</p>
    93→          <h3 className="text-sm font-medium line-clamp-2 leading-tight min-h-[2.5rem]">{product.name}</h3>
    94→          <div className="flex items-center gap-1">
    95→            <StarRating rating={product.rating} size={12} />
    96→            <span className="text-xs text-muted-foreground">({product.reviewCount})</span>
    97→          </div>
    98→          <div className="flex items-baseline gap-2 mt-auto">
    99→            <span className="text-lg font-bold text-foreground">{formatCurrency(product.price)}</span>
   100→            {product.compareAtPrice && product.compareAtPrice > product.price && (
   101→              <span className="text-sm text-muted-foreground line-through">{formatCurrency(product.compareAtPrice)}</span>
   102→            )}
   103→          </div>
   104→        </CardContent>
   105→        <CardFooter className="p-3 pt-0">
   106→          <Button size="sm" className="w-full bg-orange-500 hover:bg-orange-600 text-white" disabled={product.stock === 0} onClick={(e) => { e.stopPropagation(); addItem({ productId: product.id, name: product.name, price: product.price, image: mainImage || '', vendorName: product.vendor?.businessName || '', vendorId: product.vendorId, stock: product.stock }); toast.success('Added to cart!'); }}>
   107→            <ShoppingCart size={14} className="mr-1.5" /> Add to Cart
   108→          </Button>
   109→        </CardFooter>
   110→      </Card>
   111→    </motion.div>
   112→  );
   113→}

   120→  }
   121→  if (!products.length) {
   122→    return <div className="text-center py-16"><Package size={64} className="mx-auto text-muted-foreground/30 mb-4" /><h3 className="text-lg font-medium text-muted-foreground">No products found</h3><p className="text-sm text-muted-foreground mt-1">Try adjusting your filters or search</p></div>;
   123→  }
   124→  return <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">{products.map(p => <ProductCard key={p.id} product={p} />)}</div>;
   125→}

   238→function CategoriesNav({ mobile }: { mobile?: boolean }) {
   239→  const { data: categories } = useQuery({ queryKey: ['categories'], queryFn: () => fetch('/api/categories').then(r => r.json()).then((r: ApiResponse<Category[]>) => r.data || []) });
   240→  const { setSelectedCategory, setCustomerView, setMobileMenuOpen } = useNavigationStore();
   241→
   242→  const handleCategoryClick = (slug: string) => {
   243→    setSelectedCategory(slug);
   244→    setCustomerView('products');
   245→    if (mobile) setMobileMenuOpen(false);
   246→  };
   247→
   248→  if (mobile) {
   249→    return categories?.map(cat => (
   250→      <div key={cat.id} className="space-y-1">
   251→        <Button variant="ghost" className="w-full justify-start font-medium" onClick={() => handleCategoryClick(cat.slug)}>{cat.name}</Button>
   252→        {cat.children?.map(child => (
   253→          <Button key={child.id} variant="ghost" className="w-full justify-start pl-8 text-sm text-muted-foreground" onClick={() => handleCategoryClick(child.slug)}>{child.name}</Button>
   254→        ))}
   255→      </div>
   256→    ));
   257→  }
   258→
   259→  return <>
   260→    <Button variant="ghost" size="sm" className="h-10 text-sm font-medium" onClick={() => { setCustomerView('products'); setSelectedCategory(null); }}>All Products</Button>
   261→    {categories?.map(cat => (
   262→      <DropdownMenu key={cat.id}>
   263→        <DropdownMenuTrigger asChild>
   264→          <Button variant="ghost" size="sm" className="h-10 text-sm font-medium hover:text-orange-500">{cat.name}{cat.children?.length ? <ChevronDown size={14} className="ml-1" /> : null}</Button>
   265→        </DropdownMenuTrigger>
   266→        {cat.children?.length ? (
   267→          <DropdownMenuContent>
   268→            {cat.children.map(child => (
   269→              <DropdownMenuItem key={child.id} onClick={() => handleCategoryClick(child.slug)}>{child.name} <Badge variant="secondary" className="ml-auto text-xs">{child._count?.products || 0}</Badge></DropdownMenuItem>

  1112→        login(data.data.user, data.data.token, data.data.vendorId);

  1118→    } catch { toast.error('Something went wrong'); }

  1120→  };
  1121→
  1122→  const handleRegister = async (e: React.FormEvent) => {
  1123→    e.preventDefault();

  1127→      const res = await fetch('/api/auth', { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(form) });

  1129→      if (data.success) { login(data.data.user, data.data.token); toast.success('Account created!'); navigateTo('home'); }
