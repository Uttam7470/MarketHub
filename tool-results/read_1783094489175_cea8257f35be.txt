    38→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    39→  const stars = [];
    40→  for (let i = 1; i <= 5; i++) {
    41→    if (i <= Math.floor(rating)) stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    42→    else if (i - 0.5 <= rating) stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} style={{ clipPath: 'inset(0 50% 0 0)' }} />);
    43→    else stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    44→  }
    45→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    46→}

   243→function VendorProducts() {
   244→  const { vendorId } = useAuthStore();
   245→  const { setSelectedProductId, setVendorView } = useNavigationStore();
   246→  const qc = useQueryClient();
   247→  const [page, setPage] = useState(1);
   248→  const [search, setSearch] = useState('');
   249→  const [statusFilter, setStatusFilter] = useState<string>('all');
   250→  const [deleteId, setDeleteId] = useState<string | null>(null);
   251→  const [importOpen, setImportOpen] = useState(false);
   252→  const [importFile, setImportFile] = useState<File | null>(null);
   253→  const [importing, setImporting] = useState(false);
   254→  const [confirmImport, setConfirmImport] = useState(false);
   255→
   256→  const { data, isLoading } = useQuery({
   257→    queryKey: ['vendor-products', vendorId, page, search, statusFilter],
   258→    queryFn: () => {
   259→      const params = new URLSearchParams({ vendorId: vendorId || '', page: String(page), limit: '12', search });
   260→      if (statusFilter !== 'all') params.set('status', statusFilter);
   261→      return fetch(`/api/products?${params}`).then(r => r.json()).then((r: ApiResponse<Product[]>) => r);
   262→    },
   263→    enabled: !!vendorId,
   264→  });
   265→
   266→  const deleteMutation = useMutation({
   267→    mutationFn: (id: string) => fetch(`/api/products?id=${id}`, { method: 'DELETE' }).then(r => r.json()),
   268→    onSuccess: () => { qc.invalidateQueries({ queryKey: ['vendor-products'] }); toast.success('Product deleted'); setDeleteId(null); },
   269→    onError: () => toast.error('Failed to delete product'),
   270→  });
   271→
   272→  const duplicateMutation = useMutation({
   273→    mutationFn: (id: string) => fetch(`/api/vendor/products/${id}/duplicate`, { method: 'POST' }).then(r => r.json()),
   274→    onSuccess: (data) => {
   275→      if (data.success) { toast.success('Product duplicated successfully'); qc.invalidateQueries({ queryKey: ['vendor-products'] }); }
   276→      else toast.error(data.error || 'Failed to duplicate');
   277→    },
   278→    onError: () => toast.error('Failed to duplicate product'),
   279→  });
   280→
   281→  const handleBulkImport = async () => {
   282→    if (!importFile || !vendorId) return;
   283→    setImporting(true);
   284→    try {
   285→      const text = await importFile.text();
   286→      const lines = text.split('\n').filter(l => l.trim());
   287→      if (lines.length < 2) { toast.error('CSV file is empty or has no data rows'); setImporting(false); return; }
   288→      const headers = lines[0].split(',').map(h => h.trim().toLowerCase());
   289→      const products = [];
   290→      for (let i = 1; i < lines.length; i++) {
   291→        const values = lines[i].split(',').map(v => v.trim());
   292→        const row: Record<string, string> = {};
   293→        headers.forEach((h, idx) => { row[h] = values[idx] || ''; });
   294→        products.push(row);
   295→      }
   296→      const res = await fetch('/api/vendor/products/bulk-import', {
   297→        method: 'POST',
   298→        headers: { 'Content-Type': 'application/json' },
   299→        body: JSON.stringify({ vendorId, products }),
   300→      });
   301→      const result = await res.json();
   302→      if (result.success) {
   303→        const d = result.data;
   304→        toast.success(`Imported ${d.successCount}/${d.totalProducts} products successfully`);
   305→        if (d.errors.length > 0) toast.error(`${d.errors.length} errors occurred`);
   306→        qc.invalidateQueries({ queryKey: ['vendor-products'] });
   307→        setImportOpen(false);
   308→        setImportFile(null);
   309→      } else {
   310→        toast.error(result.error || 'Import failed');
   311→      }
   312→    } catch {
   313→      toast.error('Failed to parse CSV file');
   314→    }
   315→    setImporting(false);
   316→  };
   317→
   318→  const downloadTemplate = () => {
   319→    const csv = 'name,description,price,compareAtPrice,stock,sku,categoryId,brandId,weight\nSample Product,Sample description,999,1299,50,SKU001,cat_id_here,brand_id_here,0.5\n';
   320→    const blob = new Blob([csv], { type: 'text/csv' });
   321→    const url = URL.createObjectURL(blob);
   322→    const a = document.createElement('a');
   323→    a.href = url; a.download = 'product_import_template.csv'; a.click();
   324→    URL.revokeObjectURL(url);
   325→  };
   326→
   327→  const handleBulkExport = async () => {

   329→      const params = new URLSearchParams({ vendorId: vendorId || '', page: '1', limit: '9999', search: '' });
   330→      const res = await fetch(`/api/products?${params}`);
   331→      const json = await res.json();
   332→      const products: Product[] = json.data || [];
   333→      if (products.length === 0) { toast.error('No products to export'); return; }
   334→      const headers = ['name', 'sku', 'price', 'stock', 'status', 'category'];
   335→      const rows = products.map(p => [
   336→        `"${p.name}"`, `"${p.sku || ''}"`, String(p.price), String(p.stock),
   337→        p.productStatus, `"${p.category?.name || ''}"`
   338→      ].join(','));
   339→      const csv = [headers.join(','), ...rows].join('\n');
   340→      const blob = new Blob([csv], { type: 'text/csv' });
   341→      const url = URL.createObjectURL(blob);
   342→      const a = document.createElement('a');
   343→      a.href = url; a.download = `products_export_${new Date().toISOString().slice(0, 10)}.csv`; a.click();
   344→      URL.revokeObjectURL(url);
   345→      toast.success(`Exported ${products.length} products`);
   346→    } catch {
   347→      toast.error('Failed to export products');
   348→    }
   349→  };
   350→
   351→  const statusColor: Record<string, string> = {
   352→    PUBLISHED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   353→    DRAFT: 'bg-gray-100 text-gray-700 dark:bg-gray-900/30 dark:text-gray-400',
   354→    ARCHIVED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   355→  };
   356→
   357→  const statusFilters = ['all', 'PUBLISHED', 'DRAFT', 'ARCHIVED'];
   358→
   359→  return (
   360→    <div className="p-6 space-y-6">
   361→      <div className="flex flex-wrap items-center justify-between gap-4">
   362→        <div><h1 className="text-2xl font-bold">Products</h1><p className="text-muted-foreground text-sm">Manage your product catalog</p></div>
   363→        <div className="flex items-center gap-2 flex-wrap">
   364→          <Button className="bg-orange-500 hover:bg-orange-600" onClick={() => setVendorView('vendor-add-product')}><Plus size={16} className="mr-1.5" />Add Product</Button>
   365→          <Button variant="outline" onClick={() => setImportOpen(true)}><Upload size={16} className="mr-1.5" />Bulk Import</Button>
   366→          <Button variant="outline" onClick={handleBulkExport}><Download size={16} className="mr-1.5" />Bulk Export</Button>
   367→        </div>
   368→      </div>
   369→
   370→      <div className="flex flex-col sm:flex-row gap-3">
   371→        <div className="relative flex-1 max-w-md"><Search size={16} className="absolute left-3 top-1/2 -translate-y-1/2 text-muted-foreground" /><Input placeholder="Search products..." value={search} onChange={e => { setSearch(e.target.value); setPage(1); }} className="pl-9" /></div>
   372→        <div className="flex gap-2 flex-wrap">
   373→          {statusFilters.map(s => (
   374→            <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-orange-500 hover:bg-orange-600' : ''} onClick={() => { setStatusFilter(s); setPage(1); }}>
   375→              {s === 'all' ? 'All' : s}
   376→            </Button>
   377→          ))}
   378→        </div>
   379→      </div>
   380→
   381→      {isLoading ? <div className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 gap-4">{Array.from({length: 6}).map((_, i) => <Skeleton key={i} className="h-40 rounded-xl" />)}</div> :
