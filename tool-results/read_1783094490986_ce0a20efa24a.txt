    51→function useRequireAuth() {
    52→  const { isAuthenticated } = useAuthStore();
    53→  const { navigateTo } = useNavigationStore();
    54→  return (action: string) => {
    55→    if (!isAuthenticated) {
    56→      toast.warning('Login required', { description: `Please log in to ${action}.` });
    57→      navigateTo('login');
    58→      return false;
    59→    }
    60→    return true;
    61→  };
    62→}
    63→const discountPercent = (price: number, compare?: number | null) => compare ? Math.round(((compare - price) / compare) * 100) : 0;
    64→const badgeColors: Record<string, string> = { BEST_SELLER: 'bg-emerald-500 text-white', NEW_ARRIVAL: 'bg-blue-500 text-white', LIMITED_TIME: 'bg-red-500 text-white', FESTIVAL_OFFER: 'bg-amber-500 text-white' };
    65→const badgeLabels: Record<string, string> = { BEST_SELLER: 'Best Seller', NEW_ARRIVAL: 'New', LIMITED_TIME: 'Limited', FESTIVAL_OFFER: 'Festival' };

    67→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    68→  const stars = [];
    69→  for (let i = 1; i <= 5; i++) {
    70→    if (i <= Math.floor(rating)) {
    71→      stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    72→    } else if (i - 0.5 <= rating) {
    73→      stars.push(<StarHalf key={i} className="fill-amber-400 text-amber-400" size={size} />);
    74→    } else {
    75→      stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    76→    }
    77→  }
    78→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    79→}

    81→function ProductCard({ product }: { product: Product }) {
    82→  const { navigateTo, setSelectedProductId } = useNavigationStore();
    83→  const { addItem } = useCartStore();
    84→  const { toggleItem, hasItem } = useWishlistStore();
    85→  const { toggleItem: toggleCompare, hasItem: hasCompare } = useCompareStore();
    86→  const requireAuth = useRequireAuth();
    87→  const isWishlisted = hasItem(product.id);
    88→  const isCompared = hasCompare(product.id);
    89→  const discount = discountPercent(product.price, product.compareAtPrice);
    90→  const mainImage = product.images?.[0]?.url;
    91→
    92→  const handleAddToCart = (e: React.MouseEvent) => {
    93→    e.stopPropagation();
    94→    if (!requireAuth('add items to cart')) return;
    95→    addItem({ productId: product.id, name: product.name, price: product.price, image: mainImage || '', vendorName: product.vendor?.businessName || '', vendorId: product.vendorId, stock: product.stock });
    96→    toast.success('Added to cart', { description: `${product.name} has been added to your cart.` });
    97→  };
    98→
    99→  const handleWishlistToggle = (e: React.MouseEvent) => {
   100→    e.stopPropagation();
   101→    if (!requireAuth('add items to wishlist')) return;
   102→    toggleItem(product.id);
   103→    if (isWishlisted) {
   104→      toast.info('Removed from wishlist');
   105→    } else {
   106→      toast.success('Added to wishlist', { description: product.name });
   107→    }
   108→  };
   109→
   110→  const handleCompareToggle = (e: React.MouseEvent) => {
   111→    e.stopPropagation();
   112→    if (!requireAuth('compare products')) return;
   113→    toggleCompare(product.id);
   114→    if (isCompared) {
   115→      toast.info('Removed from comparison');
   116→    } else {
   117→      toast.success('Added to comparison');
   118→    }
   119→  };
   120→
   121→  return (
   122→    <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }} whileHover={{ y: -4 }}>
   123→      <Card role="article" aria-label={`${product.name}, ${formatCurrency(product.price)}`} className="group overflow-hidden border-border/50 hover:shadow-lg transition-all duration-300 cursor-pointer h-full flex flex-col">
   124→        <div className="relative aspect-square overflow-hidden bg-muted/30">
   125→          {mainImage ? (
   126→            <img src={mainImage} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" loading="lazy" />
   127→          ) : (
   128→            <div className="w-full h-full flex items-center justify-center text-muted-foreground"><Package size={48} /></div>
   129→          )}
   130→          <div className="absolute top-2 left-2 flex flex-col gap-1">
   131→            {discount > 0 && <Badge className="bg-red-500 text-white text-xs">{discount}% OFF</Badge>}
   132→            {product.badge && (
   133→              <Badge className={badgeColors[product.badge] || 'bg-gray-500 text-white'}>{badgeLabels[product.badge]}</Badge>
   134→            )}
   135→          </div>
   136→          {product.stock < 5 && product.stock > 0 && <Badge variant="secondary" className="absolute top-2 right-10 text-xs">Low Stock</Badge>}
   137→          {product.stock === 0 && <div className="absolute inset-0 bg-black/50 flex items-center justify-center"><Badge variant="destructive">Out of Stock</Badge></div>}
   138→          <div className="absolute top-2 right-2 flex flex-col gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   139→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" aria-label={isWishlisted ? `Remove ${product.name} from wishlist` : `Add ${product.name} to wishlist`} onClick={handleWishlistToggle}>
   140→              <Heart size={14} className={isWishlisted ? 'fill-red-500 text-red-500' : ''} />
   141→            </Button>
   142→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={handleCompareToggle}>
   143→              <GitCompare size={14} className={isCompared ? 'text-orange-500' : ''} />
   144→            </Button>
   145→          </div>
   146→        </div>
   147→        <CardContent className="p-3 flex-1 flex flex-col gap-1.5" onClick={() => { setSelectedProductId(product.id); navigateTo('product-detail'); }}>
   148→          <p className="text-xs text-muted-foreground truncate">{product.vendor?.businessName}</p>
   149→          <h3 className="text-sm font-medium line-clamp-2 leading-tight min-h-[2.5rem]">{product.name}</h3>
   150→          <div className="flex items-center gap-1">
   151→            <StarRating rating={product.rating} size={12} />
   152→            <span className="text-xs text-muted-foreground">({product.reviewCount})</span>
   153→          </div>
   154→          <div className="flex items-baseline gap-2 mt-auto">
   155→            <span className="text-lg font-bold text-foreground">{formatCurrency(product.price)}</span>
   156→            {product.compareAtPrice && product.compareAtPrice > product.price && (
   157→              <span className="text-sm text-muted-foreground line-through">{formatCurrency(product.compareAtPrice)}</span>
   158→            )}
   159→          </div>
   160→        </CardContent>
   161→        <CardFooter className="p-3 pt-0">
   162→          <Button size="sm" className="w-full bg-orange-500 hover:bg-orange-600 text-white" aria-label={`Add ${product.name} to cart`} disabled={product.stock === 0} onClick={handleAddToCart}>
   163→            <ShoppingCart size={14} className="mr-1.5" /> Add to Cart
   164→          </Button>
   165→        </CardFooter>
   166→      </Card>
   167→    </motion.div>
   168→  );
   169→}

  1075→function ReviewsSection({ productId, reviews, reviewCount, rating }: { productId: string; reviews: any[]; reviewCount: number; rating: number }) {
  1076→  const { isAuthenticated, user } = useAuthStore();
  1077→  const queryClient = useQueryClient();
  1078→  const requireAuth = useRequireAuth();
  1079→  const [showForm, setShowForm] = useState(false);
  1080→  const [form, setForm] = useState({ rating: 5, title: '', comment: '', images: '' });
  1081→  const [sortBy, setSortBy] = useState('latest');
  1082→  const sortedReviews = useMemo(() => {
  1083→    const r = [...reviews];
  1084→    switch(sortBy) {
  1085→      case 'highest': return r.sort((a,b) => b.rating - a.rating);
  1086→      case 'lowest': return r.sort((a,b) => a.rating - b.rating);
  1087→      case 'helpful': return r.sort((a,b) => (b.helpfulCount||0) - (a.helpfulCount||0));
  1088→      default: return r.sort((a,b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
  1089→    }
  1090→  }, [reviews, sortBy]);
  1091→
  1092→  const submitReview = useMutation({
  1093→    mutationFn: async () => {
  1094→      const res = await fetch('/api/reviews', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: user!.id, productId, rating: form.rating, title: form.title, comment: form.comment, images: form.images.trim() || undefined }) });
  1095→      return res.json();
  1096→    },
  1097→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['product', productId] }); toast.success('Review submitted!'); setShowForm(false); setForm({ rating: 5, title: '', comment: '', images: '' }); },
  1098→    onError: () => toast.error('Failed to submit review'),
  1099→  });
  1100→
  1101→  return (
  1102→    <div className="space-y-6">
  1103→      <div className="flex items-center justify-between">
  1104→        <div className="flex items-center gap-4">
  1105→          <div className="text-center"><div className="text-4xl font-bold">{rating}</div><StarRating rating={rating} size={16} /><p className="text-sm text-muted-foreground mt-1">{reviewCount} reviews</p></div>
  1106→        </div>
  1107→        {isAuthenticated && <Button variant="outline" onClick={() => { if (!requireAuth('write a review')) return; setShowForm(!showForm); }}>{showForm ? 'Cancel' : 'Write a Review'}</Button>}
  1108→      </div>
  1109→
  1110→      {showForm && (
  1111→        <Card className="p-4 space-y-4">
