   266→function AdminVendors() {
   267→  const confirm = useConfirm();
   268→  const [statusFilter, setStatusFilter] = useState('all');
   269→  const qc = useQueryClient();
   270→  const [detailVendor, setDetailVendor] = useState<Vendor | null>(null);
   271→  const [rejectDialogOpen, setRejectDialogOpen] = useState(false);
   272→  const [rejectVendorId, setRejectVendorId] = useState<string | null>(null);
   273→  const [rejectionReason, setRejectionReason] = useState('');
   274→  const [showWallet, setShowWallet] = useState(false);
   275→
   276→  const { data: walletData } = useQuery({
   277→    queryKey: ['admin-vendor-wallets'],
   278→    queryFn: () => fetch('/api/admin/vendors/wallets').then(r => r.json()).then((r: any) => r.data || []),
   279→  });
   280→
   281→  const params = new URLSearchParams({ limit: '50' });
   282→  if (statusFilter !== 'all') params.set('status', statusFilter);
   283→
   284→  const { data, isLoading } = useQuery({
   285→    queryKey: ['admin-vendors', statusFilter],
   286→    queryFn: () => fetch(`/api/vendors?${params}`).then(r => r.json()).then((r: any) => r),
   287→  });
   288→
   289→  const updateVendor = useMutation({
   290→    mutationFn: ({ id, status, rejectionReason: reason }: { id: string; status: string; rejectionReason?: string }) =>
   291→      fetch(`/api/vendors/${id}`, {
   292→        method: 'PUT',
   293→        headers: { 'Content-Type': 'application/json' },
   294→        body: JSON.stringify({ status, ...(reason ? { rejectionReason: reason } : {}) }),
   295→      }).then(r => r.json()),
   296→    onSuccess: () => {
   297→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   298→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });
   299→      toast.success('Vendor updated');
   300→    },
   301→    onError: () => toast.error('Failed to update vendor'),
   302→  });
   303→
   304→  const handleReject = () => {
   305→    if (!rejectVendorId) return;
   306→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   307→    setRejectDialogOpen(false);
   308→    setRejectVendorId(null);
   309→    setRejectionReason('');
   310→    setDetailVendor(null);
   311→  };
   312→
   313→  const openRejectDialog = (vendorId: string) => {
   314→    setRejectVendorId(vendorId);
   315→    setRejectionReason('');
   316→    setRejectDialogOpen(true);
   317→  };
   318→
   319→  const statusColor: Record<string, string> = {
   320→    PENDING: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   321→    APPROVED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   322→    SUSPENDED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   323→    REJECTED: 'bg-gray-100 text-gray-700 dark:bg-gray-800 dark:text-gray-300',
   324→  };
   325→
   326→  const pendingCount = data?.data?.filter((v: Vendor) => v.status === 'PENDING').length || 0;
   327→
   328→  return (
   329→    <div className="p-6 space-y-6">
   330→      <div className="flex items-center justify-between flex-wrap gap-3">
   331→        <div>
   332→          <h1 className="text-2xl font-bold">Vendor Management</h1>
   333→          <p className="text-muted-foreground text-sm">Manage vendor accounts and applications</p>
   334→        </div>
   335→        {pendingCount > 0 && (
   336→          <Badge className="bg-amber-500 text-white px-3 py-1 text-sm cursor-pointer" onClick={() => setStatusFilter('PENDING')}>
   337→            <Clock size={14} className="mr-1.5" />{pendingCount} Pending Review
   338→          </Badge>
   339→        )}
   340→      </div>
   341→
   342→      <div className="flex gap-2 flex-wrap">
   343→        {['all', 'PENDING', 'APPROVED', 'SUSPENDED', 'REJECTED'].map(s => (
   344→          <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-amber-600' : ''} onClick={() => setStatusFilter(s)}>
   345→            {s === 'all' ? 'All' : s.charAt(0) + s.slice(1).toLowerCase()}
   346→            {s === 'PENDING' && pendingCount > 0 ? ` (${pendingCount})` : ''}
   347→          </Button>
   348→        ))}
   349→      </div>
   350→
   351→      {isLoading ? <div className="space-y-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-20 rounded-xl" />)}</div> :
   352→        <Card>
   353→          <div className="overflow-x-auto">
