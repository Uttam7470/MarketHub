   262→function AdminVendors() {
   263→  const [statusFilter, setStatusFilter] = useState('all');
   264→  const qc = useQueryClient();
   265→  const [detailVendor, setDetailVendor] = useState<Vendor | null>(null);
   266→  const [rejectDialogOpen, setRejectDialogOpen] = useState(false);
   267→  const [rejectVendorId, setRejectVendorId] = useState<string | null>(null);
   268→  const [rejectionReason, setRejectionReason] = useState('');
   269→  const [showWallet, setShowWallet] = useState(false);
   270→
   271→  const { data: walletData } = useQuery({
   272→    queryKey: ['admin-vendor-wallets'],
   273→    queryFn: () => fetch('/api/admin/vendors/wallets').then(r => r.json()).then((r: any) => r.data || []),
   274→  });
   275→
   276→  const params = new URLSearchParams({ limit: '50' });
   277→  if (statusFilter !== 'all') params.set('status', statusFilter);
   278→
   279→  const { data, isLoading } = useQuery({
   280→    queryKey: ['admin-vendors', statusFilter],
   281→    queryFn: () => fetch(`/api/vendors?${params}`).then(r => r.json()).then((r: any) => r),
   282→  });
   283→
   284→  const updateVendor = useMutation({
   285→    mutationFn: ({ id, status, rejectionReason: reason }: { id: string; status: string; rejectionReason?: string }) =>
   286→      fetch(`/api/vendors/${id}`, {
   287→        method: 'PUT',
   288→        headers: { 'Content-Type': 'application/json' },
   289→        body: JSON.stringify({ status, ...(reason ? { rejectionReason: reason } : {}) }),
   290→      }).then(r => r.json()),
   291→    onSuccess: () => {
   292→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   293→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });
   294→      toast.success('Vendor updated');
   295→    },
   296→    onError: () => toast.error('Failed to update vendor'),
   297→  });
   298→
   299→  const handleReject = () => {
   300→    if (!rejectVendorId) return;
   301→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   302→    setRejectDialogOpen(false);
   303→    setRejectVendorId(null);
   304→    setRejectionReason('');
   305→    setDetailVendor(null);
   306→  };
   307→
   308→  const openRejectDialog = (vendorId: string) => {
   309→    setRejectVendorId(vendorId);
   310→    setRejectionReason('');
   311→    setRejectDialogOpen(true);
   312→  };
   313→
   314→  const statusColor: Record<string, string> = {
   315→    PENDING: 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400',
   316→    APPROVED: 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400',
   317→    SUSPENDED: 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400',
   318→    REJECTED: 'bg-gray-100 text-gray-700 dark:bg-gray-800 dark:text-gray-300',
   319→  };
   320→
   321→  const pendingCount = data?.data?.filter((v: Vendor) => v.status === 'PENDING').length || 0;
   322→
   323→  return (
   324→    <div className="p-6 space-y-6">
   325→      <div className="flex items-center justify-between flex-wrap gap-3">
   326→        <div>
   327→          <h1 className="text-2xl font-bold">Vendor Management</h1>
   328→          <p className="text-muted-foreground text-sm">Manage vendor accounts and applications</p>
   329→        </div>
   330→        {pendingCount > 0 && (
   331→          <Badge className="bg-amber-500 text-white px-3 py-1 text-sm cursor-pointer" onClick={() => setStatusFilter('PENDING')}>
   332→            <Clock size={14} className="mr-1.5" />{pendingCount} Pending Review
   333→          </Badge>
   334→        )}
   335→      </div>
   336→
   337→      <div className="flex gap-2 flex-wrap">
   338→        {['all', 'PENDING', 'APPROVED', 'SUSPENDED', 'REJECTED'].map(s => (
   339→          <Button key={s} variant={statusFilter === s ? 'default' : 'outline'} size="sm" className={statusFilter === s ? 'bg-amber-600' : ''} onClick={() => setStatusFilter(s)}>
   340→            {s === 'all' ? 'All' : s.charAt(0) + s.slice(1).toLowerCase()}
   341→            {s === 'PENDING' && pendingCount > 0 ? ` (${pendingCount})` : ''}
   342→          </Button>
   343→        ))}
   344→      </div>
   345→
   346→      {isLoading ? <div className="space-y-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-20 rounded-xl" />)}</div> :
   347→        <Card>
   348→          <div className="overflow-x-auto">
   349→            <Table>
   350→              <TableHeader>
   351→                <TableRow>
   352→                  <TableHead>Vendor</TableHead>
   353→                  <TableHead className="hidden md:table-cell">Email</TableHead>
   354→                  <TableHead className="hidden lg:table-cell">GST/PAN</TableHead>
   355→                  <TableHead>Products</TableHead>
