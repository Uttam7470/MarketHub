   246→      {/* Tabs: Description, Specs, Reviews, Q&A */}
   247→      <div className="mt-10">
   248→        <Tabs value={activeTab} onValueChange={setActiveTab}>

   258→              ))}</TableBody></Table></div>

   260→          </TabsContent>

   300→function Table({ children, className }: { children: React.ReactNode; className?: string }) {
   301→  return <div className={`rounded-lg border overflow-hidden ${className || ''}`}><table className="w-full text-sm">{children}</table></div>;
   302→}
   303→function TableBody({ children }: { children: React.ReactNode }) { return <tbody>{children}</tbody>; }
   304→function TableRow({ children, className }: { children: React.ReactNode; className?: string }) { return <tr className={`border-b last:border-0 ${className || ''}`}>{children}</tr>; }
   305→function TableCell({ children, className }: { children: React.ReactNode; className?: string }) { return <td className={`px-4 py-3 ${className || ''}`}>{children}</td>; }

   307→function CustomersAlsoBought({ productId }: { productId: string }) {
   308→  const { data: products = [], isLoading } = useQuery({
   309→    queryKey: ['customers-also-bought', productId],
   310→    queryFn: () => fetch(`/api/products?limit=8&exclude=${productId}`).then(r => r.json()).then((r: any) => (r.data || []).sort(() => Math.random() - 0.5).slice(0, 4)),
   311→    enabled: !!productId,
   312→  });
   313→  if (isLoading) return <div className="grid grid-cols-2 md:grid-cols-4 gap-4">{Array.from({length:4}).map((_,i)=><Skeleton key={i} className="aspect-square rounded-xl" />)}</div>;
   314→  if (!products.length) return null;
   315→  return <div className="grid grid-cols-2 md:grid-cols-4 gap-4">{products.map(p => <ProductCard key={p.id} product={p} />)}</div>;
   316→}

   318→function ReviewsSection({ productId, reviews, reviewCount, rating }: { productId: string; reviews: any[]; reviewCount: number; rating: number }) {
   319→  const { isAuthenticated, user } = useAuthStore();
   320→  const queryClient = useQueryClient();
   321→  const requireAuth = useRequireAuth();
   322→  const [showForm, setShowForm] = useState(false);
   323→  const [form, setForm] = useState({ rating: 5, title: '', comment: '', images: '' });
   324→  const [sortBy, setSortBy] = useState('latest');
   325→  const sortedReviews = useMemo(() => {
   326→    const r = [...reviews];
   327→    switch(sortBy) {
   328→      case 'highest': return r.sort((a,b) => b.rating - a.rating);
   329→      case 'lowest': return r.sort((a,b) => a.rating - b.rating);
   330→      case 'helpful': return r.sort((a,b) => (b.helpfulCount||0) - (a.helpfulCount||0));
   331→      default: return r.sort((a,b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
   332→    }
   333→  }, [reviews, sortBy]);
   334→
   335→  const submitReview = useMutation({
   336→    mutationFn: async () => {
   337→      const res = await fetch('/api/reviews', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: user!.id, productId, rating: form.rating, title: form.title, comment: form.comment, images: form.images.trim() || undefined }) });
   338→      return res.json();
   339→    },
   340→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['product', productId] }); toast.success('Review submitted!'); setShowForm(false); setForm({ rating: 5, title: '', comment: '', images: '' }); },
   341→    onError: () => toast.error('Failed to submit review'),
   342→  });
   343→
   344→  return (
   345→    <div className="space-y-6">
   346→      <div className="flex items-center justify-between">
   347→        <div className="flex items-center gap-4">
   348→          <div className="text-center"><div className="text-4xl font-bold">{rating}</div><StarRating rating={rating} size={16} /><p className="text-sm text-muted-foreground mt-1">{reviewCount} reviews</p></div>
   349→        </div>
   350→        {isAuthenticated && <Button variant="outline" onClick={() => { if (!requireAuth('write a review')) return; setShowForm(!showForm); }}>{showForm ? 'Cancel' : 'Write a Review'}</Button>}
   351→      </div>
   352→
   353→      {showForm && (
   354→        <Card className="p-4 space-y-4">
   355→          <div><Label>Rating</Label><div className="flex gap-1 mt-1">{[1,2,3,4,5].map(s => <button key={s} aria-label={`Rate ${s} out of 5 stars`} onClick={() => setForm(f => ({...f, rating: s}))}><Star size={24} className={s <= form.rating ? 'fill-amber-400 text-amber-400' : 'text-muted-foreground/30'} /></button>)}</div></div>
   356→          <div><Label>Title</Label><Input placeholder="Summary of your experience" value={form.title} onChange={e => setForm(f => ({...f, title: e.target.value}))} className="mt-1" /></div>
   357→          <div><Label>Review</Label><Textarea placeholder="Tell us more..." value={form.comment} onChange={e => setForm(f => ({...f, comment: e.target.value}))} className="mt-1" rows={3} /></div>
   358→          <div><Label>Images (URLs, one per line)</Label><Textarea placeholder="Paste image URLs here..." value={form.images} onChange={e => setForm(f => ({...f, images: e.target.value}))} className="mt-1" rows={2} /></div>
   359→          <Button className="bg-orange-500 hover:bg-orange-600" onClick={() => submitReview.mutate()} disabled={submitReview.isPending || !form.comment.trim()}>{submitReview.isPending ? 'Submitting...' : 'Submit Review'}</Button>
   360→        </Card>
   361→      )}
   362→
   363→      <div className="space-y-4">
   364→        <div className="flex items-center justify-between">
   365→          <span className="text-sm text-muted-foreground">Sort by:</span>
   366→          <Select value={sortBy} onValueChange={setSortBy}>
   367→            <SelectTrigger aria-label="Sort reviews" className="w-40 h-8 text-sm"><SelectValue /></SelectTrigger>
   368→            <SelectContent>

   390→            {review.comment && <p className="text-sm text-muted-foreground mt-1">{review.comment}</p>}

   397→            )}
   398→          </Card>

   400→      </div>
   401→    </div>
   402→  );

   405→function ProductQASection({ productId, qaList }: { productId: string; qaList: any[] }) {
   406→  const { isAuthenticated, user } = useAuthStore();
   407→  const queryClient = useQueryClient();
   408→  const [showDialog, setShowDialog] = useState(false);
   409→  const [question, setQuestion] = useState('');
   410→
   411→  const submitQuestion = useMutation({
   412→    mutationFn: async () => {
   413→      const res = await fetch(`/api/products/${productId}/qa`, {
   414→        method: 'POST',
   415→        headers: { 'Content-Type': 'application/json' },
   416→        body: JSON.stringify({ userId: user!.id, question }),
   417→      });
   418→      return res.json();
   419→    },
   420→    onSuccess: () => {
   421→      queryClient.invalidateQueries({ queryKey: ['product-qa', productId] });
   422→      toast.success('Question submitted!');
   423→      setShowDialog(false);
   424→      setQuestion('');
   425→    },
   426→    onError: () => toast.error('Failed to submit question'),
   427→  });
   428→
   429→  return (
   430→    <div className="space-y-6">
   431→      <div className="flex items-center justify-between">
   432→        <div className="flex items-center gap-2">
   433→          <HelpCircle size={20} className="text-orange-500" />
   434→          <h3 className="text-lg font-bold">Questions & Answers</h3>
   435→          <Badge variant="secondary">{qaList.length}</Badge>
   436→        </div>
   437→        <Button variant="outline" onClick={() => { if (!isAuthenticated) { toast.info('Please login to ask a question'); return; } setShowDialog(true); }}>
   438→          <Send size={14} className="mr-1.5" />Ask Question
   439→        </Button>
   440→      </div>
   441→
   442→      <Dialog open={showDialog} onOpenChange={setShowDialog}>
   443→        <DialogContent>
   444→          <DialogHeader>
   445→            <DialogTitle>Ask a Question</DialogTitle>
   446→            <DialogDescription>Get answers from the community and seller</DialogDescription>
   447→          </DialogHeader>
   448→          <div className="space-y-4 py-2">
   449→            <div>
   450→              <Label>Your Question</Label>
   451→              <Textarea placeholder="What would you like to know about this product?" value={question} onChange={e => setQuestion(e.target.value)} className="mt-1" rows={3} />
   452→            </div>
   453→          </div>
   454→          <DialogFooter>
   455→            <Button variant="outline" onClick={() => setShowDialog(false)}>Cancel</Button>
   456→            <Button className="bg-orange-500 hover:bg-orange-600" onClick={() => submitQuestion.mutate()} disabled={!question.trim() || submitQuestion.isPending}>
   457→              {submitQuestion.isPending ? 'Submitting...' : 'Submit Question'}
   458→            </Button>
   459→          </DialogFooter>
   460→        </DialogContent>
   461→      </Dialog>
   462→
   463→      <div className="space-y-4">
   464→        {qaList.length === 0 && (
   465→          <p className="text-muted-foreground text-center py-8">No questions yet. Be the first to ask!</p>
   466→        )}
   467→        {qaList.map((qa: any) => (
   468→          <Card key={qa.id} className="p-4">
   469→            <div className="flex items-start gap-3">
   470→              <Avatar className="h-8 w-8 shrink-0"><AvatarFallback className="text-xs">{qa.user?.name?.[0] || 'U'}</AvatarFallback></Avatar>
   471→              <div className="flex-1 min-w-0">
   472→                <div className="flex items-center gap-2">
   473→                  <p className="font-medium text-sm">{qa.user?.name || 'Anonymous'}</p>
   474→                  <span className="text-xs text-muted-foreground">{new Date(qa.createdAt).toLocaleDateString()}</span>
   475→                </div>
   476→                <p className="text-sm mt-1">{qa.question}</p>
   477→                {qa.answer && (
   478→                  <div className="mt-3 ml-4 pl-4 border-l-2 border-orange-500 bg-orange-50 dark:bg-orange-900/10 rounded-r-lg p-3">
   479→                    <div className="flex items-center gap-2 mb-1">
   480→                      <span className="text-xs font-medium text-orange-600 dark:text-orange-400">Seller Answer</span>
   481→                    </div>
   482→                    <p className="text-sm text-muted-foreground">{qa.answer}</p>
   483→                  </div>
   484→                )}
   485→                {!qa.answer && (
   486→                  <p className="text-xs text-muted-foreground mt-1 italic">No answer yet</p>
   487→                )}
   488→              </div>
   489→            </div>
   490→          </Card>
   491→        ))}
   492→      </div>
   493→    </div>
   494→  );
   495→}
