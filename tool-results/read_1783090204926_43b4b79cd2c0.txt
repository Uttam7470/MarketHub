    51→function useRequireAuth() {
    52→  const { isAuthenticated } = useAuthStore();
    53→  const { navigateTo } = useNavigationStore();
    54→  return (action: string) => {
    55→    if (!isAuthenticated) {
    56→      toast.warning('Login required', { description: `Please log in to ${action}.` });
    57→      navigateTo('login');
    58→      return false;
    59→    }
    60→    return true;
    61→  };
    62→}
    63→const discountPercent = (price: number, compare?: number | null) => compare ? Math.round(((compare - price) / compare) * 100) : 0;
    64→const badgeColors: Record<string, string> = { BEST_SELLER: 'bg-emerald-500 text-white', NEW_ARRIVAL: 'bg-blue-500 text-white', LIMITED_TIME: 'bg-red-500 text-white', FESTIVAL_OFFER: 'bg-amber-500 text-white' };
    65→const badgeLabels: Record<string, string> = { BEST_SELLER: 'Best Seller', NEW_ARRIVAL: 'New', LIMITED_TIME: 'Limited', FESTIVAL_OFFER: 'Festival' };

    67→function StarRating({ rating, size = 16 }: { rating: number; size?: number }) {
    68→  const stars = [];
    69→  for (let i = 1; i <= 5; i++) {
    70→    if (i <= Math.floor(rating)) {
    71→      stars.push(<Star key={i} className="fill-amber-400 text-amber-400" size={size} />);
    72→    } else if (i - 0.5 <= rating) {
    73→      stars.push(<StarHalf key={i} className="fill-amber-400 text-amber-400" size={size} />);
    74→    } else {
    75→      stars.push(<Star key={i} className="text-muted-foreground/30" size={size} />);
    76→    }
    77→  }
    78→  return <span className="inline-flex items-center gap-0.5">{stars}</span>;
    79→}

    81→function ProductCard({ product }: { product: Product }) {
    82→  const { navigateTo, setSelectedProductId } = useNavigationStore();
    83→  const { addItem } = useCartStore();
    84→  const { toggleItem, hasItem } = useWishlistStore();
    85→  const { toggleItem: toggleCompare, hasItem: hasCompare } = useCompareStore();
    86→  const requireAuth = useRequireAuth();
    87→  const isWishlisted = hasItem(product.id);
    88→  const isCompared = hasCompare(product.id);
    89→  const discount = discountPercent(product.price, product.compareAtPrice);
    90→  const mainImage = product.images?.[0]?.url;
    91→
    92→  const handleAddToCart = (e: React.MouseEvent) => {
    93→    e.stopPropagation();
    94→    if (!requireAuth('add items to cart')) return;
    95→    addItem({ productId: product.id, name: product.name, price: product.price, image: mainImage || '', vendorName: product.vendor?.businessName || '', vendorId: product.vendorId, stock: product.stock });
    96→    toast.success('Added to cart', { description: `${product.name} has been added to your cart.` });
    97→  };
    98→
    99→  const handleWishlistToggle = (e: React.MouseEvent) => {
   100→    e.stopPropagation();
   101→    if (!requireAuth('add items to wishlist')) return;
   102→    toggleItem(product.id);
   103→    if (isWishlisted) {
   104→      toast.info('Removed from wishlist');
   105→    } else {
   106→      toast.success('Added to wishlist', { description: product.name });
   107→    }
   108→  };
   109→
   110→  const handleCompareToggle = (e: React.MouseEvent) => {
   111→    e.stopPropagation();
   112→    toggleCompare(product.id);
   113→    if (isCompared) {
   114→      toast.info('Removed from comparison');
   115→    } else {
   116→      toast.success('Added to comparison');
   117→    }
   118→  };
   119→
   120→  return (
   121→    <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3 }} whileHover={{ y: -4 }}>
   122→      <Card role="article" aria-label={`${product.name}, ${formatCurrency(product.price)}`} className="group overflow-hidden border-border/50 hover:shadow-lg transition-all duration-300 cursor-pointer h-full flex flex-col">
   123→        <div className="relative aspect-square overflow-hidden bg-muted/30">
   124→          {mainImage ? (
   125→            <img src={mainImage} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" loading="lazy" />
   126→          ) : (
   127→            <div className="w-full h-full flex items-center justify-center text-muted-foreground"><Package size={48} /></div>
   128→          )}
   129→          <div className="absolute top-2 left-2 flex flex-col gap-1">
   130→            {discount > 0 && <Badge className="bg-red-500 text-white text-xs">{discount}% OFF</Badge>}
   131→            {product.badge && (
   132→              <Badge className={badgeColors[product.badge] || 'bg-gray-500 text-white'}>{badgeLabels[product.badge]}</Badge>
   133→            )}
   134→          </div>
   135→          {product.stock < 5 && product.stock > 0 && <Badge variant="secondary" className="absolute top-2 right-10 text-xs">Low Stock</Badge>}
   136→          {product.stock === 0 && <div className="absolute inset-0 bg-black/50 flex items-center justify-center"><Badge variant="destructive">Out of Stock</Badge></div>}
   137→          <div className="absolute top-2 right-2 flex flex-col gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
   138→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" aria-label={isWishlisted ? `Remove ${product.name} from wishlist` : `Add ${product.name} to wishlist`} onClick={handleWishlistToggle}>
   139→              <Heart size={14} className={isWishlisted ? 'fill-red-500 text-red-500' : ''} />
   140→            </Button>
   141→            <Button size="icon" variant="secondary" className="h-8 w-8 rounded-full shadow-md" onClick={handleCompareToggle}>
   142→              <GitCompare size={14} className={isCompared ? 'text-orange-500' : ''} />
   143→            </Button>
   144→          </div>
   145→        </div>
   146→        <CardContent className="p-3 flex-1 flex flex-col gap-1.5" onClick={() => { setSelectedProductId(product.id); navigateTo('product-detail'); }}>
   147→          <p className="text-xs text-muted-foreground truncate">{product.vendor?.businessName}</p>
   148→          <h3 className="text-sm font-medium line-clamp-2 leading-tight min-h-[2.5rem]">{product.name}</h3>
   149→          <div className="flex items-center gap-1">
   150→            <StarRating rating={product.rating} size={12} />
   151→            <span className="text-xs text-muted-foreground">({product.reviewCount})</span>
   152→          </div>
   153→          <div className="flex items-baseline gap-2 mt-auto">
   154→            <span className="text-lg font-bold text-foreground">{formatCurrency(product.price)}</span>
   155→            {product.compareAtPrice && product.compareAtPrice > product.price && (
   156→              <span className="text-sm text-muted-foreground line-through">{formatCurrency(product.compareAtPrice)}</span>
   157→            )}
   158→          </div>
   159→        </CardContent>
   160→        <CardFooter className="p-3 pt-0">
   161→          <Button size="sm" className="w-full bg-orange-500 hover:bg-orange-600 text-white" aria-label={`Add ${product.name} to cart`} disabled={product.stock === 0} onClick={handleAddToCart}>
   162→            <ShoppingCart size={14} className="mr-1.5" /> Add to Cart
   163→          </Button>
   164→        </CardFooter>
   165→      </Card>
   166→    </motion.div>
   167→  );
   168→}

   170→function ProductGrid({ products, loading }: { products: Product[]; loading?: boolean }) {
   171→  if (loading) {
   172→    return <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
   173→      {Array.from({ length: 8 }).map((_, i) => <Skeleton key={i} className="aspect-square rounded-xl" />)}
   174→    </div>;
   175→  }
   176→  if (!products.length) {
   177→    return (
   178→      <div className="flex flex-col items-center justify-center py-16 text-center">
   179→        <SearchX size={64} className="text-muted-foreground/30 mb-4" />
   180→        <h3 className="text-lg font-medium text-muted-foreground">No products found</h3>
   181→        <p className="text-sm text-muted-foreground mt-1">Try adjusting your filters or search</p>
   182→      </div>
   183→    );
   184→  }
   185→  return <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">{products.map(p => <ProductCard key={p.id} product={p} />)}</div>;
   186→}
