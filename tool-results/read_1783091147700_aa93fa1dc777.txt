   291→   291→    onSuccess: () => {
   292→   292→      qc.invalidateQueries({ queryKey: ['admin-vendors'] });
   293→   293→      qc.invalidateQueries({ queryKey: ['admin-dashboard'] });

   297→   297→  });

   300→   300→    if (!rejectVendorId) return;
   301→   301→    updateVendor.mutate({ id: rejectVendorId, status: 'REJECTED', rejectionReason: rejectionReason || 'Application does not meet our requirements.' });
   302→   302→    setRejectDialogOpen(false);
   303→   303→    setRejectVendorId(null);
   304→   304→    setRejectionReason('');
   305→   305→    setDetailVendor(null);
   306→   306→  };
